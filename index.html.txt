    <script>
        let cart = [];
        let total = 0;
const sheetURL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQGP488InXYjOP4SbgDmeNgPapkgFyMpjWr0ShvACi56yOHNEY3VsdLP3tEHXuFXc3FRRjgaFb2QktM/pub?output=csv';
// دالة لجلب البيانات من جوجل شيت وعرضها
async function loadProductsFromSheet() {
    try {
        // بنطلب البيانات من الرابط بصيغة CSV
        const response = await fetch(sheetURL);
        const data = await response.text();
        
        // تقسيم البيانات لصفوف وتجاهل صف العناوين الأول
        const rows = data.split('\n').slice(1);
        const container = document.getElementById('products-container');
        
        // تنظيف المكان قبل وضع المنتجات الجديدة
        container.innerHTML = '';

        rows.forEach(row => {
            // تقسيم الصف لأعمدة (الاسم، السعر، الصورة)
            const columns = row.split(',');
            
            // التأكد إن الصف فيه بيانات كافية
            if (columns.length >= 3) {
                const name = columns[0].trim();
                const price = columns[1].trim();
                const image = columns[2].trim();

                // إضافة كارد المنتج في الصفحة
                container.innerHTML += `
                    <div class="col-6 col-md-4">
                        <div class="product-card shadow-sm text-center">
                            <div class="product-img-wrapper">
                                <img src="${image}" alt="${name}">
                                <button class="btn-add-cart" onclick="addToCart('${name}', ${price})">إضافة للسلة</button>
                            </div>
                            <div class="p-3">
                                <h5 class="fw-bold">${name}</h5>
                                <p class="text-secondary">${price} EGP</p>
                            </div>
                        </div>
                    </div>
                `;
            }
        });
    } catch (error) {
        console.error('حدث خطأ أثناء تحميل المنتجات:', error);
    }
}

// تشغيل الدالة بمجرد تحميل الصفحة
window.addEventListener('load', loadProductsFromSheet);
        function toggleCart() {
            document.getElementById('cartSidebar').classList.toggle('active');
            document.getElementById('overlay').style.display = 
                document.getElementById('overlay').style.display === 'block' ? 'none' : 'block';
        }

        function addToCart(name, price) {
            cart.push({name, price});
            total += price;
            updateCartUI();
            alert('تم إضافة المنتج لسلتك!');
        }

        function updateCartUI() {
            const list = document.getElementById('cart-items-list');
            const count = document.getElementById('cart-count');
            const totalEl = document.getElementById('cart-total');
            
            count.innerText = cart.length;
            totalEl.innerText = total + " EGP";
            
            list.innerHTML = cart.map((item, index) => `
                <div class="d-flex justify-content-between align-items-center mb-3 border-bottom pb-2">
                    <div>
                        <p class="mb-0 fw-bold">${item.name}</p>
                        <small class="text-secondary">${item.price} EGP</small>
                    </div>
                    <button class="btn btn-sm text-danger" onclick="removeFromCart(${index})">
                        <i class="fas fa-trash"></i>
                    </button>
                </div>
            `).join('');
        }// دالة لحذف منتج معين من السلة
        function removeFromCart(index) {
            total -= cart[index].price; // طرح السعر من الإجمالي
            cart.splice(index, 1); // إزالة العنصر من المصفوفة
            updateCartUI(); // تحديث الشكل
        }

        // دالة إرسال الطلب للواتساب
        function checkoutWhatsApp() {
            if (cart.length === 0) {
                alert("سلتك فارغة!");
                return;
            }

            let phoneNumber = "201011590702"; // ضع رقمك هنا مفتاح الدولة ثم الرقم
            let message = "مرحباً Eclaté Egypt، أريد طلب المنتجات التالية:%0a%0a";
            
            cart.forEach((item, i) => {
                message += `${i + 1}- ${item.name} (${item.price} EGP)%0a`;
            });

            message += `%0a*الإجمالي: ${total} EGP*`;
            message += "%0a%0aهل المنتجات متوفرة؟";

            // فتح رابط الواتساب
            window.open(`https://wa.me/${phoneNumber}?text=${message}`, '_blank');
        }
